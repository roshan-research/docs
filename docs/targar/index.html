    import requests

values = {
    "text": "سلام",
    "translateFrom": "persian",
    "translateTo": "english"
}

headers = {
  'Authorization': 'Token TOKEN_KEY',
}
response = requests.post('http://targar.ir/api/translate/', data=values, headers=headers)

print(response.json())
